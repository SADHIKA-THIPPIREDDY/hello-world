<script type="text/javascript">
text_truncate = function(str, len, end) {
    if (len == null) {
      len= 100;
    }
    if (end == null) {
      end= '...';
    }
    if (str.length > len) {
      return str.substring(0, len - end.length) + end;
    } else {
      return str;
    }
  }
</script>
